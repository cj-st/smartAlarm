define(["./CustomMarker", "sizzle"],

  function (customMarker, q) {

    tomtom.TrafficMarker = tomtom.CustomMarker.extend({

      /**
       * A marker which is used to mark traffic incidents and clusters of traffic incidents.
       * @constructor
       * @class TrafficMarker
       * @namespace tomtom
       * @extends tomtom.CustomMarker
       * @param {L.LatLng} latlng The lat / lng the marker should be added at.  Can either pass an array, [lat, lng] or a L.LatLng object.
       * @param {Object} incident The incident information returned from the traffic service.
       */
      initialize: function (latlng, incident) {

        this.incidentType = tomtom.TrafficMarker.getTypeString(incident.ty);
        this.iconCategory = tomtom.TrafficMarker.getCategoryString(incident.ic);
        this.clusterSize = incident.cs;
        this._incident = incident;

        // sometimes, clusters aren't returned as category == 13
        if (this.clusterSize && this.iconCategory != "region") {
          this.iconCategory = "cluster";
        }

        tomtom.CustomMarker.prototype.initialize.apply(this, [latlng, "trafficIcon"]);

      },

      onAdd: function (map) {
        /* jshint unused:false */
        tomtom.CustomMarker.prototype.onAdd.apply(this, arguments);

        this.configureIcon();

        this.off("mouseover", this._onMouseOver);
        this.off("mouseout", this._onMouseOut);
        this.on("mouseover", this._onMouseOver, this);
        this.on("mouseout", this._onMouseOut, this);

      },

      onRemove: function (map) {
        /* jshint unused:false */

        tomtom.CustomMarker.prototype.onRemove.apply(this, arguments);

        this.off("mouseover", this._onMouseOver);
        this.off("mouseout", this._onMouseOut);

      },

      configureIcon: function () {

        var category = this.iconCategory,
          type = this.incidentType;

        // set up the background
        this.setBackgroundImage("traffic-icons/" + category + "-" + type + ".png");

        // set up the element's id
        this._icon.id = "traffic_incident_" + this._incident.id;

        // set up the offset
        if (category == "cluster" || category == "region") {

          L.DomUtil.addClass(this._icon, "trafficIcon-cluster");

          this._icon.innerHTML = this.clusterSize;

          if (category == "region") {
            L.DomUtil.addClass(this._icon, "trafficIcon-region");
          }

          this._configureClusterPopup();

        } else {

          this._configurePopup();

        }

      },

      setBackgroundImage: function (image) {

        this._icon.style.backgroundImage = "url(\"" + tomtom.baseImagePath + image + "\")";

      },

      _onMouseOut: function () {

        L.DomUtil.removeClass(this._icon, "trafficIcon-hover");

      },

      _onMouseOver: function () {

        L.DomUtil.addClass(this._icon, "trafficIcon-hover");

      },

      _configurePopup: function () {

        this.bindPopup(this._getIncidentMarkup(this._incident));

      },

      _getIncidentMarkup: function (incident) {
        var baseUrl = tomtom.baseImagePath;
        var bundle = new tomtom.StringBundle("IncidentItem");

        // default properties to empty
        var data = L.Util.extend({
          r: "",
          dl: "--",
          l: "--",
          d: "",
          f: "--",
          t: "--",
          icon: "url(" + baseUrl + "traffic-icons/" + tomtom.TrafficMarker.getCategoryString(incident.ic) + "-" + tomtom.TrafficMarker.getTypeString(incident.ty) + ".png)",
          "labelDelay": bundle.getString("delay"),
          "labelLength": bundle.getString("length"),
          "labelFrom": bundle.getString("from"),
          "labelTo": bundle.getString("to")
        }, incident);

        // format the delay time
        if (data.dl && !isNaN(data.dl)) {
          data.dl = Math.round(data.dl / 60);
          data.dl += " " + tomtom.StringBundle.getString("units.time.minute", null, data.dl);
        }

        // format the distance
        if (data.l && !isNaN(data.l)) {
          data.l = tomtom.StringBundle.getString("units.distance", null, data.l);
        }

        return L.Util.template(INCIDENT_MARKUP, data);
      },

      _configureClusterPopup: function () {

        var baseUrl = tomtom.baseImagePath;
        var poiCount = 0;
        var poiMarkup = "";
        var bundle = new tomtom.StringBundle("IncidentBalloon.Cluster");
        var incident = this._incident;

        // sort the POIs by the longest delay
        var pois = this.pois = incident.cpoi.sort(function (a, b) {
          if (a.dl > b.dl) {
            return -1;
          } else if (a.dl == b.dl) {
            return 0;
          } else {
            return 1;
          }
        });

        this._pois = pois;

        // generate the markup for the top 4 POIs
        for (var poiIndex = 0; poiIndex < pois.length && poiIndex < 4; poiIndex++) {
          var poi = pois[poiIndex];
          var className = "";

          // trim the from / to fields
          if (poi.f && poi.f.length > CLUSTER_ROAD_NAME_MAX_LENGTH) {
            poi.f = poi.f.substring(0, CLUSTER_ROAD_NAME_MAX_LENGTH) + "...";
          }

          if (poi.t && poi.t.length > CLUSTER_ROAD_NAME_MAX_LENGTH) {
            poi.t = poi.t.substring(0, CLUSTER_ROAD_NAME_MAX_LENGTH) + "...";
          }

          if (poiIndex + 1 == pois.length) {
            className = "last";
          }

          var data = L.Util.extend({
            r: "",
            dl: "--",
            l: "--",
            d: "",
            f: "--",
            t: "--",
            index: poiIndex,
            className: className,
            icon: "url(" + baseUrl + "traffic-icons/" + tomtom.TrafficMarker.getCategoryString(poi.ic) + "-" + tomtom.TrafficMarker.getTypeString(poi.ty) + ".png)"
          }, poi);

          // format the delay time
          if (data.dl && !isNaN(data.dl)) {
            data.dl = Math.round(data.dl / 60);
            data.dl += " " + tomtom.StringBundle.getString("units.time.minute", null, data.dl);
          }

          // format the distance
          if (data.l && !isNaN(data.l)) {
            data.l = tomtom.StringBundle.getString("units.distance", null, data.l);
          }

          poiMarkup += L.Util.template(CLUSTER_ROW_MARKUP, data);

          poiCount++;
        }

        // generate the main markup
        var markup = L.Util.template(CLUSTER_MARKUP, {
          incidentCount: this._incident.cs,
          labelIncidents: bundle.getString("title"),
          mostSevereIncidents: bundle.getString("listHeader", { count: poiCount }, poiCount),
          orderedBy: bundle.getString("orderedByDelay"),
          incidents: poiMarkup
        });

        this.bindPopup(markup);

      },

      openPopup: function () {
        /* jshint -W083*/
        tomtom.CustomMarker.prototype.openPopup.apply(this, arguments);

        // create a function to handle the click event for clusters
        var clusterRowClick = L.Util.bind(function (e, element) {

          var poiIndex = parseInt(element.getAttribute("data-poi-index"), 10),
            incident = null,
            pois = this._pois;

          if (!isNaN(poiIndex)) {
            incident = pois[poiIndex];
          }

          if (incident) {
            this._showIncidentPopup(incident);
          }

          // make sure the event doesn't bubble up, if it did the popup would close instantly
          L.DomEvent.stopPropagation(e);

        }, this);

        // bind to the click events of the cluster rows
        var elements = q(".trafficInfo-cluster tr", this._popup._contentNode);

        for (var elementIndex = 0; elementIndex < elements.length; elementIndex++) {

          var element = elements[elementIndex];

          L.DomEvent.on(element, "click", function (e) {
            clusterRowClick(e, this);
          });

        }

      },

      _showIncidentPopup: function (incident) {
        var anchor = L.point(this.options.popupAnchor) || new L.Point(0, 0);

        anchor = anchor.add(L.Popup.prototype.options.offset);

        var options = {offset: anchor};
        var popup = new L.Popup(options, this).setContent(this._getIncidentMarkup(incident));
        popup.setLatLng(this._latlng);
        this._map.openPopup(popup);

        return popup;
      }

    });

    tomtom.TrafficMarker.getTypeString = function (incidentType) {
      // determine the name of the severity
      switch (incidentType) {
        case 1:
          return "minor";
        case 2:
          return "moderate";
        case 3:
          return "major";
        default:
          return "unknown";
      }
    };

    tomtom.TrafficMarker.getCategoryString = function (incidentCategory) {

      // determine the name of the incident category
      switch (incidentCategory) {
        case 1:
          return "accdnt";
        case 4:
          return "rain";
        case 5:
          return "ice";
        case 2:
        case 3:
        case 10:
        case 11:
        case 12:
          return "block";
        case 7:
        case 8:
          return "closed";
        case 9:
          return "rworks";
        case 13:
          return "cluster";
        case 999:
          return "region";
        default:
          return "traffic";
      }
    };

    var INCIDENT_MARKUP = "<div class=\"trafficInfo\">" +
      "<div class=\"header\">" +
      "<div style=\"background-image: {icon}\" class=\"icon\"></div>" +
      "<div class=\"road\">{r}</div>" +
      "<div class=\"delay\"><label>{labelDelay}:</label>{dl}</div>" +
      "<div class=\"length\"><label>{labelLength}:</label>{l}</div>" +
      "</div>" +
      "<div class=\"body\">" +
      "<div class=\"description\">{d}</div>" +
      "<div class=\"location\">" +
      "<div class=\"line\"><label>{labelFrom}:</label><div>{f}</div></div>" +
      "<div class=\"line\"><label>{labelTo}:</label><div>{t}</div></div>" +
      "</div></div></div>";

    var CLUSTER_MARKUP = "<div class=\"trafficInfo trafficInfo-cluster\">" +
      "<div class=\"header\">{labelIncidents} {incidentCount}</div>" +
      "<h3>{mostSevereIncidents} <span>{orderedBy}</span></h3>" +
      "<table>{incidents}</table>" +
      "</div>";

    var CLUSTER_ROW_MARKUP = "<tr class=\"{className}\" data-poi-index=\"{index}\">" +
      "<td><div class=\"icon\" style=\"background-image: {icon}\"></div></td>" +
      "<td class=\"location\"><div class=\"from\">{f}</div><div class=\"to\">{t}</div></td>" +
      "<td class=\"delay\">{dl}</td>" +
      "<td class=\"length\">{l}</td>";

    var CLUSTER_ROAD_NAME_MAX_LENGTH = 18;

    return tomtom.TrafficMarker;

  }
);

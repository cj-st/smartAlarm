define([
    "../BaseObject",
    "../Utils"
  ],
  function (base) {

    tomtom.dom.DomUtilResult = base.extend({

      /**
       * Wraps a series of DOM Nodes and provides extra functionality.
       * This class should not be instantiated directly, instead use the methods of the {{#crossLink "tomtom.dom.DomUtil"}}DomUtil{{/crossLink}} class.
       * @constructor
       * @class DomUtilResult
       * @namespace tomtom.dom
       * @param {Array} elements The elements to wrap.
       * @param {tomtom.dom.DomUtil} dutil The DomUtil object to use to interface with Dojo / jQuery.
       */
      initialize: function (elements, du) {

        this.elements = elements;
        this.du = du;

      },

      /**
       * Gets or sets the HTML of the contained elements.
       * When getting HTML, the first element contained within this DomUtilResult will be used.
       * @method html
       * @param {String} [value] The value to set.
       * @returns {String|tomtom.dom.DomUtilResult}
       * @chainable
       */
      html: function (value) {
        // if a value was not supplied, return the inner HTML
        if (value === null || value === undefined) {
          var first = this.first();

          if (first) {
            return first.innerHTML;
          } else {
            return null;
          }
        } else {
          this.du.html(this.elements, value);

          return this;
        }
      },

      /**
       * Gets or sets a css property for the contained elements.
       * When getting a css property, the first contained element will be used.
       * @method css
       * @param {String} property The css property to get or set.
       * @param {String} [value] The value to set.
       * @returns {String|tomtom.dom.DomUtilResult}
       * @chainable
       */
      css: function (property, value) {
        if (value === null || value === undefined) {
          var first = this.first();

          if (first) {
            return this.du.css(first, property);
          } else {
            return null;
          }
        } else {
          this.each(this.hitch(function (element) {
            this.du.css(element, property, value);
          }));
        }
      },

      /**
       * Appends the specified HTMl or DOM node to the contained elements.
       * Note that if a DOM Node is passed, it will be added to the first contained element.
       * DOM nodes can't be added in more than one location.
       * @method append
       * @param {String|DOM Node} value The HTML or DOM Node to append to the contained elements.
       * @returns {tomtom.dom.DomUtilResult}
       * @chainable
       */
      append: function (value) {
        if (typeof(value) == "string") {
          this.each(function (element) {
            element.innerHTML += value;
          });
        } else {
          var first = this.first();

          if (first) {

            this.du.each(value, function (element) {
              first.appendChild(element);
            });

          }
        }

        return this;
      },

      /**
       * Gets the first contained element.
       * @method first
       * @returns {DOM Node}
       */
      first: function () {
        return this.get(0);
      },

      /**
       * Gets the element at the specified index.
       * Note if index is not specified the first element is returned.
       * @method get
       * @param {Number} [index] The index of the element to retrieve.
       * @returns {DOM Node}
       */
      get: function (index) {
        // if an index was not supplied, use the first element
        if (index === null || index === undefined) {
          index = 0;
        }

        if (this.elements.length > index) {
          return this.elements[index];
        } else {
          return null;
        }
      },

      /**
       * Executes the specified function for each contained element.
       * @method each
       * @param {Function} callback The callback to execute for each element.
       *  The element and the element index will be passed as parameters respectively.
       * @returns {tomtom.dom.DomUtilResult}
       * @chainable
       */
      each: function (callback, scope) {

        this.du.each(this.elements, callback, scope);

        return this;
      },

      /**
       * Gets or sets the form value for the contained elements.
       * When getting, the value of the first element is returned.
       * @param {String} [value] The value to set.
       * @method val
       * @returns {tomtom.dom.DomUtilResult}
       * @chainable
       */
      val: function (value) {
        if (value === undefined || value === null) {
          var first = this.first();

          if (first && first.value) {
            return first.value;
          }

          return "";
        } else {
          this.du.val(this.elements, value);
          return this;
        }
      },

      /**
       * Searches all of the children (recursively) using the specified query.
       * @method find
       * @param {String} query The query used to filter all children.
       * @returns {tomtom.dom.DomUtilResult}
       * @chainable
       */
      find: function (query) {
        return this.du.query(query, this.elements);
      },

      /**
       * Attaches an event handler to the contained elements.
       * @method on
       * @param {String} event The name of the event to bind to.
       * @param {Function} callback The callback to execute when the event is triggered.
       * @param {Object} [scope] The context in which to execute the callback.
       * @returns {Object}
       */
      on: function (event, callback, scope) {
        return this.du.on(this.elements, event, callback, scope);
      },

      /**
       * Detaches an event handler from the contained elements.
       * @method off
       * @param {String} event The name of the event to bind to.
       * @param {Function} callback The callback to execute when the event is triggered.
       * @returns {Object}
       */
      off: function (event, callback) {
        return this.du.off(this.elements, event, callback);
      },

      /**
       * Gets or sets an attribute for the contained elements.
       * When getting, the attribute of the first element is returned.
       * @method attr
       * @param {String} name The name of the attribute to get or set.
       * @param {String} [value] The value to set.
       * @returns {String|tomtom.dom.DomUtilResult}
       */
      attr: function (name, value) {
        return this.du.attr(this.elements, name, value);
      },

      /**
       * Gets or sets associated data for the contained elements.
       * When getting associated data, the data from the first element is returned.
       * @method data
       * @param {String} name The name of the associated data.
       * @param {Object} [value] The data to set.
       * @returns {Object|tomtom.dom.DomUtilResult}
       */
      data: function (name, value) {
        return this.du.data(this.elements, name, value);
      },

      /**
       * Removes the specified attribute from the contained elements.
       * @method removeAttr
       * @param {String} name The name of the attribute to remove.
       * @returns {tomtom.dom.DomUtilResult}
       */
      removeAttr: function (name) {
        this.du.removeAttr(this.elements, name);
        return this;
      },

      /**
       * Gets the next sibling of the first contained element.
       * @method next
       * @param {String} filter The filter to apply to the siblings.
       * @returns {tomtom.dom.DomUtilResult}
       */
      next: function (filter) {
        return this.du.next(this.first(), filter);
      },

      /**
       * Gets the previous sibling of the first contained element.
       * @method prev
       * @param {String} filter The filter to apply to the siblings.
       * @returns {tomtom.dom.DomUtilResult}
       */
      prev: function (filter) {
        return this.du.prev(this.first(), filter);
      },

      /**
       * Gets the children of the contained elements.
       * @method children
       * @param {String} filter The filter to apply to the children.
       * @returns {tomtom.dom.DomUtilResult}
       */
      children: function (filter) {
        return this.du.children(this.elements, filter);
      },

      /**
       * Determines how many elements are contained.
       * @method length
       * @returns
       */
      length: function () {
        return this.elements.length;
      },

      /**
       * Adds a class to all of the contained elements.
       * @method addClass
       * @param {String} className The name of the class to add.
       * @returns {tomtom.dom.DomUtilResult}
       */
      addClass: function (className) {
        this.du.addClass(this.elements, className);
        return this;
      },

      /**
       * Removes the specified class from the contained elements.
       * @method removeClass
       * @param {String} className The name of the class to remove.
       * @returns {tomtom.dom.DomUtilResult}
       */
      removeClass: function (className) {
        this.du.removeClass(this.elements, className);
        return this;
      },

      /**
       * Gets offset information for the first contained element.
       * The returned object will have width / height properties that specify the offset from the
       * element's offset parent.
       * @method offset
       * @returns {Object}
       */
      offset: function () {
        return this.du.offset(this.first());
      },

      /**
       * Gets the positioning information for the first contained element.
       * The returned object will have width / height properties that specify the offset from the edges of the window.
       * @method position
       * @returns {Object}
       */
      position: function () {
        return this.du.position(this.first());
      },

      /**
       * Gets the offsetHeight of the first contained element.
       * @method offsetHeight
       * @returns {Number}
       */
      offsetHeight: function () {
        var first = this.first();

        if (first) {
          return first.offsetHeight;
        }

        return null;
      },

      /**
       * Gets the offsetWidth of the first contained element.
       * @method offsetWidth
       * @returns {Number}
       */
      offsetWidth: function () {
        var first = this.first();

        if (first) {
          return first.offsetWidth;
        }

        return null;
      },

      /**
       * Gets the height of the first contained element.
       * @method height
       * @return {Number} The height of the first contained element.
       */
      height: function () {

        var first = this.first();

        if (first) {
          return first.clientHeight;
        }

        return null;

      },

      /**
       * Gets the width of the first contained element.
       * @method width
       * @return {Number} The width of the first contained element.
       */
      width: function () {

        var first = this.first();

        if (first) {
          return first.clientWidth;
        }

        return null;

      },

      /**
       * Clears everything from all of the contained elements.
       * @return {DomUtilResult}
       * @chainable
       */
      empty: function () {

        this.each(function (element) {

          this.du.empty(element);

        }, this);

        return this;

      },

      /**
       * Hides all of the contained elements, setting the CSS display property to "none".
       * @return {DomUtilResult}
       * @chainable
       */
      hide: function () {

        this.each(function () {
          this.css("display", "none");
        }, this);

        return this;

      },

      /**
       * Shows all of the contained elements, setting the CSS display property to "block";
       * @return {DomUtilResult}
       * @chainable
       */
      show: function () {

        this.each(function () {
          this.css("display", "block");
        }, this);

        return this;

      },

      /**
       * Gets all parents of all contained elements.
       * @return {DomUtilResult}
       */
      parents: function () {

        return this.du.parents(this);

      },

      /**
       * Determines whether or not any of the contained elements have the specified className.
       * @param {String} className The className to search for.
       * @return {Boolean} whether or not the specified className exists on any contained element.
       */
      hasClass: function (className) {

        return this.du.hasClass(this, className);

      }

    });

    return tomtom.dom.DomUtilResult;

  }
);

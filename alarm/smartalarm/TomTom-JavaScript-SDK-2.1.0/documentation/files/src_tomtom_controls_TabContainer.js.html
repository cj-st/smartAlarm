define([
  "../dom/DomUtil",
  "text!../templates/controls/TabContainer.html"
], function (du, tabContainerTemplate) {

  tomtom.controls.TabContainer = L.Class.extend({

    includes: L.Mixin.Events,

    /**
     * A simple tab container control.
     * @example
     *  var tabContainer = new tomtom.controls.TabContainer({ domNode: "tabs" });
     *  tabContainer.addTab({ content: "My custom tab", title: "My Tab", tag: "mytab" });
     * @constructor
     * @class TabContainer
     * @namespace tomtom.controls
     * @param {Object} options Configuration options for the tab control.
     * @param {String | DOM Node} domNode The DOM node which will contain the tab control.
     * @param {Array} tabs A list of tab options (see {{#crossLink "TabContainer/addTab:method"}}addTab{{/crossLink}}).
     */
    initialize: function (options) {

      this._selectedIndex = 0;

      this._domNode = du.create(L.DomUtil.get(options.domNode));
      this._domNode.html(tabContainerTemplate);
      this._domNode.addClass("tt-tab-container");

      this._header = this._domNode.find(".tt-tab-container-header");
      this._body = this._domNode.find(".tt-tab-container-body");

      if (options.tabs) {

        options.tabs.forEach(function (tabData) {

          this.addTab(tabData);

        }, this);

      }

      this.resize();

    },

    /**
     * Adds a tab based on the options supplied.
     * @example
     *  tabContainer.addTab({ content: "My tab content", title: "My Tab", tag: "mytab" });
     * @method addTab
     * @param {Object} options Options used to configure the tab being added.
     * @param {String | DOM Node} options.content The content that will be appended to the body of the tab.
     * @param {String | DOM Node} options.title The title that will be displayed in the tab container's tab bar.
     * @param {String} tag A unique identifier for the tab.
     */
    addTab: function (options) {

      // create the tab
      var tab = du.create("div", "tt-tab-outer").attr("data-tag", options.tag);
      var tabInner = du.create("div", "tt-tab-inner").html(options.title);

      tab.append(tabInner);
      this._header.append(tab);

      // set up click event
      L.DomEvent.on(tab.get(0), "click", function () {
        this.setSelectedTab(options.tag);
      }, this);

      // create the body
      var body = du.create("div", "tt-tab-body").html(options.content).attr("data-tag", options.tag);
      this._body.append(body);

      this._updateSelected();

    },

    /**
     * Updates the tab that has the specified tag.
     * @method updateTab
     * @param {String} tag The tag of the tab to update.
     * @param {Object} options The options of the tab to change. See {{#crossLink "TabContainer/addTab:method"}}addTab{{/crossLink}}.
     */
    updateTab: function (tag, options) {

      var header = this._getHeader(tag);
      var body = this._getBody(tag);

      if (header && body) {

        header.find(".tt-tab-inner").html(options.title);
        body.html(options.content);

      }

    },

    /**
     * Resizes the tab container based on the height of its container.
     * @method resize
     */
    resize: function () {

      var offset = this._body.offsetHeight() - this._body.height();

      this._body.css("height", (this._domNode.offsetHeight() - this._header.offsetHeight() - this._header.offset().top - offset) + "px");

    },

    /**
     * Sets the selected tab.
     * @method setSelectedTab
     * @param {String} tag The tag of the tab to select.
     */
    setSelectedTab: function (tag) {

      this._header.find(".tt-tab-selected").removeClass("tt-tab-selected");
      this._getHeader(tag).addClass("tt-tab-selected");

      this._body.children().css("display", "none");
      this._getBody(tag).css("display", "block");

    },

    _getHeader: function (tag) {

      return this._header.find(".tt-tab-outer[data-tag='" + tag + "']");

    },

    _getBody: function (tag) {

      return this._body.find(".tt-tab-body[data-tag='" + tag + "']");

    },

    _updateSelected: function () {

      this.setSelectedTab(this._header.find(".tt-tab-outer:nth-child(" + (this._selectedIndex + 1) + ")").attr("data-tag"));
      this._header.find(".tt-tab-outer-last").removeClass("tt-tab-outer-last");
      this._header.find(".tt-tab-outer:last-child").addClass("tt-tab-outer-last");

    }

  });

  return tomtom.controls.TabContainer;

});

define([
    "../Utils",
    "../Main"
  ],

  function () {

    tomtom.layers.WMSLayer = L.TileLayer.WMS.extend({

      /**
       * An extension of the L.TileLayer.WMS which adds support for custom srs parameter.
       * @constructor
       * @class WMSLayer
       * @namespace tomtom.layers
       * @param {String} url The URL to the WMS layer.
       * @param {Object} options WMS Layer options.
       * @param {String} options.layers='' Comma-separated list of WMS layers to show.
       * @param {String} [options.styles=''] Comma-separated list of WMS styles.
       * @param {String} [options.format='image/jpeg'] WMS image format (use 'image/png' for layers with transparency).
       * @param {Boolean} [options.transparent=false] If true, the WMS service will return images with transparency.
       * @param {String} [options.version='1.1.1'] Version of the WMS service to use.
       */
      initialize: function (url, options) {
        L.TileLayer.WMS.prototype.initialize.apply(this, arguments);

        if (options.srs){
          this.wmsParams.srs = options.srs;
        }
      },

      onAdd: function (map) {

        var projectionKey = parseFloat(this.wmsParams.version) >= 1.3 ? 'crs' : 'srs';

        if (!this.options.srs){
          this.wmsParams[projectionKey] = map.options.crs.code;
        }

        L.TileLayer.prototype.onAdd.call(this, map);

      }

    });
  }
);

define([
    "./Main",
    "./Marker",
    "./dom/DomUtil",
    "./CustomDivIcon"
  ],

  function (main, Marker, du, CustomDivIcon) {

    tomtom.CustomMarker = tomtom.Marker.extend({

      /**
       * An extension of the Marker class which allows the use of custom icons and therefore allows CSS customization.
       * This class can be extended to build custom markers that need to be customized via CSS.
       * For instance, the {{#crossLink "tomtom.TrafficMarker"}}TrafficMarker{{/crossLink}} class uses this as a base.
       * @constructor
       * @class CustomMarker
       * @extends tomtom.Marker
       * @param {L.LatLng} The lat / lng the marker should be added at.  Can either pass an array, [lat, lng] or a L.LatLng object.
       * @param {String} className The css class that will be added to the marker element.
       * @param {Object} [options] The options to pass to the tomtom.Marker constructor (markerOptions parameter).  Also, an html property can be specified as an option
       *  to set the initial HTML of the marker div.
       */
      initialize: function (latlng, className, options) {

        if (!options) {
          options = {};
        }

        // set up the icon
        options.icon = new CustomDivIcon({ className: className, html: options.html || "&nbsp;", iconSize: null });

        tomtom.Marker.prototype.initialize.apply(this, [latlng, null, options]);

      },

      /**
       * Sets the CSS class of the icon.
       * @method setClassName
       * @param {String} className
       */
      setClassName: function (className) {

        this._icon.className = className;

      },

      /**
       * Sets the HTML content of the marker.
       * @param {String | DOM Node} html The HTML string or DOM node to set the content to.
       */
      setHTML: function (html) {

        du.html(this._icon, html);

      }

    });

    return tomtom.CustomMarker;

  }
);

define([
  "../Main",
  "../dom/DomUtil",
  "./RouteControl",
  "./TabContainer",
  "./FindLocationControl",
  "./TrafficIncidentListControl"
], function (main, du) {

  tomtom.controls.RoutePlannerControl = L.Class.extend({

    /**
     * Creates a tab container with the RouteControl, FindLocationControl and TrafficIncidentListControl added.
     * This is a simple way to create a quick route planner application.
     * @class RoutePlannerControl
     * @namespace tomtom.controls
     * @constructor
     * @param options Configuration options for the route planner control.
     * @param {tomtom.Map} options.map The map the route planner control will attach to.
     * @param {String | DOM Node} options.domNode The DOM Node that the route planner control will be addded to.
     */
    initialize: function (options) {

      this._domNode = du.create(L.DomUtil.get(options.domNode));
      this._map = options.map;

      // create DOM nodes for each tab
      this._routeNode = du.create("div");
      this._findNode = du.create("div");
      this._trafficNode = du.create("div");

      // get the "general" string bundle, which contains text for the Route, Location and Traffic tabs
      var bundle = new tomtom.StringBundle("General");

      // create the tab control, adding three tabs
      this._tabControl = new tomtom.controls.TabContainer({
        domNode: this._domNode.get(0),
        tabs: [
          { title: bundle.getString("plan_a_route"), content: this._routeNode.get(0), tag: "route" },
          { title: bundle.getString("find_a_location"), content: this._findNode.get(0), tag: "location" },
          { title: bundle.getString("traffic"), content: this._trafficNode.get(0), tag: "traffic" }
        ]
      });

      // create the route control, attach it to the DOM node created above
      this._routeControl = new tomtom.controls.RouteControl({
        domNode: this._routeNode.get(0),
        map: this._map
      });

      // create the find location control, attach it to the DOM node created above
      this._findControl = new tomtom.controls.FindLocationControl({
        domNode: this._findNode.get(0),
        map: this._map
      });

      // create the traffic incident list control, attach it to the DOM node created above
      this._trafficControl = new tomtom.controls.TrafficIncidentListControl({
        domNode: this._trafficNode.get(0),
        map: this._map
      });

      // resize the tab container when the window resizes
      L.DomEvent.on(window, "resize", this._tabControl.resize, this._tabControl);

    }

  });

  return tomtom.controls.RoutePlannerControl;

});
